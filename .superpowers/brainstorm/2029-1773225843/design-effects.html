## Effect Params ##

Outline:
  color: (R,G,B,A)    # default (0,0,0,255) black
  thickness: int       # 1-4 pixels
  mode: str            # "outer" | "inner" | "both"
  connectivity: int    # 4 (cardinal only) | 8 (diagonal too)

Drop Shadow:
  color: (R,G,B,A)     # default (0,0,0,180)
  offset_x: int        # -16 to +16, default 2
  offset_y: int        # -16 to +16, default 2
  blur: int            # 0-3, default 0 (hard pixel shadow)
  opacity: float       # 0.0-1.0, default 0.7

Inner Shadow:
  color: (R,G,B,A)     # default (0,0,0,160)
  offset_x: int        # -16 to +16, default 1
  offset_y: int        # -16 to +16, default 1
  blur: int            # 0-3, default 0
  opacity: float       # 0.0-1.0, default 0.5

Hue/Saturation Shift:
  hue: int             # -180 to +180 degrees, default 0
  saturation: float    # 0.0-2.0 multiplier, default 1.0
  lightness: int       # -100 to +100, default 0

Gradient Map:
  stops: list[(float, (R,G,B,A))]  # position 0.0-1.0 → color
  opacity: float                    # blend strength 0.0-1.0

Glow/Bloom:
  threshold: int       # 0-255, brightness cutoff, default 200
  radius: int          # 1-8 blur radius, default 3
  intensity: float     # 0.0-2.0 multiplier, default 1.0
  tint: (R,G,B,A)      # glow color tint, default (255,255,255,255)

Pattern Overlay:
  pattern: str         # "scanlines"|"crosshatch"|"dots"|"diagonal"|"noise"|"checkerboard"
  blend_mode: str      # any of the 9 existing blend modes
  opacity: float       # 0.0-1.0, default 0.3
  scale: int           # 1-4x, default 1
  offset_x: int        # pattern offset
  offset_y: int

def apply_outline(pixels, color, thickness, mode, connectivity):
    alpha = pixels[:,:,3]
    for each thickness step:
      if mode == "outer" or "both":
        # find transparent pixels adjacent to opaque (dilate - original)
        dilated = binary_dilation(alpha > 0, connectivity_struct)
        outline_mask = dilated & (alpha == 0)
        result[outline_mask] = color
      if mode == "inner" or "both":
        # find opaque pixels adjacent to transparent (original - erode)
        eroded = binary_erosion(alpha > 0, connectivity_struct)
        inner_mask = (alpha > 0) & ~eroded
        result[inner_mask] = blend(result[inner_mask], color)

def apply_drop_shadow(pixels, color, ox, oy, blur, opacity):
    # 1. Extract alpha mask
    shadow_mask = pixels[:,:,3] > 0
    # 2. Create shadow layer, fill with color at opacity
    shadow = np.zeros_like(pixels)
    shadow[shadow_mask] = [*color[:3], int(255 * opacity)]
    # 3. Offset by (ox, oy) via np.roll + zero-fill edges
    shadow = shift_array(shadow, ox, oy)
    # 4. Optional gaussian blur (scipy.ndimage or manual 3x3/5x5 kernel)
    if blur > 0: shadow = gaussian_blur(shadow, sigma=blur)
    # 5. Composite shadow BEHIND original pixels
    return alpha_composite(shadow, pixels)  # shadow first, then pixels on top

def apply_effects(pixels: np.ndarray, effects: list[LayerEffect]) -> np.ndarray:
    """Master function: applies all enabled effects in order."""
    result = pixels.copy()
    for fx in effects:
        if not fx.enabled: continue
        result = EFFECT_FUNCS[fx.type](result, **fx.params)
    return result
