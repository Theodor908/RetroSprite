def render_text(text: str, font: dict, color: tuple,
                spacing: int = 1, line_height: int = 2,
                align: str = "left") -> Image.Image:
    """Render text using a built-in bitmap font.

    Args:
        text: String to render (supports newlines)
        font: Font dict with 'char_width', 'char_height', 'glyphs'
        color: RGBA color tuple for text pixels
        spacing: Extra pixels between characters
        line_height: Extra pixels between lines
        align: 'left', 'center', or 'right'

    Returns:
        RGBA PIL Image with rendered text
    """
    if not text:
        return Image.new("RGBA", (1, 1), (0, 0, 0, 0))

    cw = font["char_width"]
    ch = font["char_height"]
    glyphs = font["glyphs"]
    lines = text.split("\n")

    # Calculate line widths
    def _line_width(line):
        if not line:
            return 0
        return len(line) * cw + (len(line) - 1) * spacing

    line_widths = [_line_width(line) for line in lines]
    img_w = max(line_widths) if line_widths else 1
    img_h = len(lines) * ch + (len(lines) - 1) * line_height
    if img_w <= 0:
        img_w = 1
    if img_h <= 0:
        img_h = 1

    img = Image.new("RGBA", (img_w, img_h), (0, 0, 0, 0))
    pixels = np.array(img)

    for row_idx, line in enumerate(lines):
        y_off = row_idx * (ch + line_height)

        # Alignment offset
        lw = line_widths[row_idx]
        if align == "center":
            x_start = (img_w - lw) // 2
        elif align == "right":
            x_start = img_w - lw
        else:
            x_start = 0

        for col_idx, char in enumerate(line):
            glyph = glyphs.get(char)
            if glyph is None:
                # Unknown char: filled rectangle
                glyph = [(1 << cw) - 1] * ch

            cx = x_start + col_idx * (cw + spacing)
            for gy, row_bits in enumerate(glyph):
                for gx in range(cw):
                    if row_bits & (1 << (cw - 1 - gx)):
                        px, py = cx + gx, y_off + gy
                        if 0 <= px < img_w and 0 <= py < img_h:
                            pixels[py, px] = color

    return Image.fromarray(pixels, "RGBA")


def get_cursor_x(text: str, cursor_pos: int, font: dict, spacing: int = 1) -> int:
    """Calculate the pixel X offset of the cursor at a given character index.

    Only considers the current line (text before last newline is ignored).
    """
    cw = font["char_width"]
    # Find position within the current line
    lines = text.split("\n")
    pos = cursor_pos
    for line in lines:
        if pos <= len(line):
            return pos * (cw + spacing)
        pos -= len(line) + 1  # +1 for the newline
    # Fallback: past end
    last_line = lines[-1] if lines else ""
    return len(last_line) * (cw + spacing)
```

def render_text_ttf(text: str, font_path: str, size: int, color: tuple,
                    spacing: int = 0, line_height: int = 0,
                    align: str = "left") -> Image.Image:
    """Render text using a TTF font via Pillow.

    Args:
        text: String to render
        font_path: Path to .ttf or .otf file
        size: Font size in pixels
        color: RGBA color tuple
        spacing: Extra pixels between characters (added to natural spacing)
        line_height: Extra pixels between lines
        align: 'left', 'center', or 'right'

    Returns:
        RGBA PIL Image with rendered text
    """
    if not text:
        return Image.new("RGBA", (1, 1), (0, 0, 0, 0))

    font = ImageFont.truetype(font_path, size)
    lines = text.split("\n")

    # Measure each line
    line_sizes = []
    for line in lines:
        if not line:
            line_sizes.append((0, size))
            continue
        bbox = font.getbbox(line)
        w = bbox[2] - bbox[0] + spacing * max(0, len(line) - 1)
        h = bbox[3] - bbox[1]
        line_sizes.append((w, max(h, size)))

    img_w = max(s[0] for s in line_sizes) if line_sizes else 1
    total_h = sum(s[1] for s in line_sizes) + line_height * max(0, len(lines) - 1)
    if img_w <= 0:
        img_w = 1
    if total_h <= 0:
        total_h = 1

    img = Image.new("RGBA", (img_w, total_h), (0, 0, 0, 0))
    draw = ImageDraw.Draw(img)

    y = 0
    for i, line in enumerate(lines):
        if not line:
            y += line_sizes[i][1] + line_height
            continue
        lw = line_sizes[i][0]
        if align == "center":
            x = (img_w - lw) // 2
        elif align == "right":
            x = img_w - lw
        else:
            x = 0

        # Draw each character individually to apply custom spacing
        if spacing > 0:
            cx = x
            for ch in line:
                draw.text((cx, y), ch, fill=color, font=font)
                bbox = font.getbbox(ch)
                cx += (bbox[2] - bbox[0]) + spacing
        else:
            draw.text((x, y), line, fill=color, font=font)

        y += line_sizes[i][1] + line_height

    return img
```

    def _on_key_press(self, event):
        """Handle keyboard input — routes to text mode when active."""
        if not self._text_mode:
            return  # Let Tkinter handle normally

        key = event.keysym
        char = event.char
        ctrl = bool(event.state & 0x0004)

        # Ctrl+Enter = commit
        if ctrl and key == "Return":
            self._exit_text_mode(commit=True)
            return "break"

        # Ctrl+T = enter transform mode with text image
        if ctrl and key.lower() == "t":
            self._text_enter_transform()
            return "break"

        # Enter = newline
        if key == "Return":
            self._text_string = (self._text_string[:self._text_cursor_pos] +
                                 "\n" + self._text_string[self._text_cursor_pos:])
            self._text_cursor_pos += 1
            self._render_text_overlay()
            return "break"

        # Escape = cancel (handled by _on_escape, but also catch here)
        if key == "Escape":
            self._exit_text_mode(commit=False)
            return "break"

        # Backspace
        if key == "BackSpace" and self._text_cursor_pos > 0:
            self._text_string = (self._text_string[:self._text_cursor_pos - 1] +
                                 self._text_string[self._text_cursor_pos:])
            self._text_cursor_pos -= 1
            self._render_text_overlay()
            return "break"

        # Delete
        if key == "Delete" and self._text_cursor_pos < len(self._text_string):
            self._text_string = (self._text_string[:self._text_cursor_pos] +
                                 self._text_string[self._text_cursor_pos + 1:])
            self._render_text_overlay()
            return "break"

        # Arrow keys
        if key == "Left" and self._text_cursor_pos > 0:
            self._text_cursor_pos -= 1
            self._render_text_overlay()
            return "break"
        if key == "Right" and self._text_cursor_pos < len(self._text_string):
            self._text_cursor_pos += 1
            self._render_text_overlay()
            return "break"

        # Home / End
        if key == "Home":
            # Move to start of current line
            before = self._text_string[:self._text_cursor_pos]
            last_nl = before.rfind("\n")
            self._text_cursor_pos = last_nl + 1
            self._render_text_overlay()
            return "break"
        if key == "End":
            # Move to end of current line
            after = self._text_string[self._text_cursor_pos:]
            nl = after.find("\n")
            if nl == -1:
                self._text_cursor_pos = len(self._text_string)
            else:
                self._text_cursor_pos += nl
            self._render_text_overlay()
            return "break"

        # Printable character
        if char and len(char) == 1 and ord(char) >= 32 and not ctrl:
            self._text_string = (self._text_string[:self._text_cursor_pos] +
                                 char + self._text_string[self._text_cursor_pos:])
            self._text_cursor_pos += 1
            self._render_text_overlay()
            return "break"

    def _text_enter_transform(self):
        """Hand off current text image to selection transform."""
        if not self._text_string.strip():
            return
        from src.selection_transform import SelectionTransform

        color = self.palette.selected_color
        font_name = self._get_text_font_name()
        img = self._render_current_text_image(color, font_name)
        if img is None:
            return

        pos = self._text_pos
        self._stop_cursor_blink()
        self._text_mode = False
        self._text_string = ""
        self.pixel_canvas.clear_floating()
        self._hide_text_context_bar()

        self._selection_transform = SelectionTransform(
            pixels=img,
            position=pos,
            source="paste",
        )
        self._draw_transform_overlay()
        self._show_transform_context_bar()
        self._update_status("Transform: drag handles, Enter=apply, Esc=cancel")

        """Open file picker to load a TTF/OTF font."""
        from tkinter import filedialog
        path = filedialog.askopenfilename(
            title="Load Font",
            filetypes=[("Font files", "*.ttf *.otf"), ("All files", "*.*")],
        )
        if path:
            import os
            name = os.path.splitext(os.path.basename(path))[0]
            self._text_loaded_fonts[name] = path
            self._text_font_var.set(name)
            # Rebuild the context bar to include new font
            self._show_text_context_bar()
            if hasattr(self, '_text_size_spin'):
                self._text_size_spin.config(state="normal")
            self._on_text_setting_change()
        else:
            # User cancelled — reset to previous font
            self._text_font_var.set("Standard 5x7")
